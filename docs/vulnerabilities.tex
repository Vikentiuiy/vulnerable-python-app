\subsection{VULN-01: CWE-89: SQL Injection}
\noindent\textbf{Location:} \texttt{src/vulnapp/injection\_vulns.py}, line 18 \quad \textbf{Rule:} \texttt{python/sql-injection} \quad \textbf{Severity:} 8.0
\\ \textbf{Reference:} \url{https://cwe.mitre.org/data/definitions/89.html}

\noindent\textbf{Planted sink:}
\begin{lstlisting}
cur.execute("SELECT * FROM users WHERE id = '" + user_id + "'")
\end{lstlisting}
\noindent\textbf{What it is.} SQL injection: untrusted input is concatenated into a SQL query string.

\noindent\textbf{Exploitation variants.}
\begin{itemize}[leftmargin=1.4em,itemsep=1pt,topsep=2pt]
\item Auth bypass: `' OR '1'='1` in a login field.
\item Data exfiltration via UNION: `' UNION SELECT username,password FROM users--`.
\item Blind/time-based extraction with `SLEEP()`/`WAITFOR`; stacked queries for write/DROP.
\end{itemize}
\noindent\textbf{Remediation.} Use parameterized queries / PreparedStatement; never build SQL by string concatenation.

\subsection{VULN-07: CWE-95: Code Injection (eval/exec)}
\noindent\textbf{Location:} \texttt{src/vulnapp/injection\_vulns.py}, line 55 \quad \textbf{Rule:} \texttt{python/code-injection} \quad \textbf{Severity:} 8.0
\\ \textbf{Reference:} \url{https://cwe.mitre.org/data/definitions/95.html}

\noindent\textbf{Planted sink:}
\begin{lstlisting}
return eval(expr)
\end{lstlisting}
\noindent\textbf{What it is.} Eval injection: untrusted input is passed to a dynamic-evaluation primitive (eval/exec/compile), so the attacker's data is executed as code.

\noindent\textbf{Exploitation variants.}
\begin{itemize}[leftmargin=1.4em,itemsep=1pt,topsep=2pt]
\item Supply `\_\_import\_\_('os').system('id')` to run arbitrary OS commands.
\item Use `().\_\_class\_\_.\_\_bases\_\_[0].\_\_subclasses\_\_()` gadget chains to escape restricted eval namespaces.
\item Read/exfiltrate secrets: `open('/etc/passwd').read()` inside the evaluated expression.
\end{itemize}
\noindent\textbf{Remediation.} Never eval/exec untrusted input. Parse with ast.literal\_eval for data, or use an explicit allow-listed dispatch table.
